function calcularIMC(peso, altura) { var imc = peso / (altura * altura); if (imc < 18.5) { return "Bajo peso"; } else if (imc >= 18.5 && imc < 24.9) { return "Peso normal"; } else if (imc >= 25 && imc < 29.9) { return "Sobrepeso"; } else if (imc >= 30 && imc < 34.9) { return "Obesidad grado I"; } else if (imc >= 35 && imc < 39.9) { return "Obesidad grado II"; } else { return "Obesidad grado III"; } } var peso = 70; var altura = 1.75; var resultadoIMC = calcularIMC(peso, altura); console.log("Tu IMC es " + resultadoIMC);
